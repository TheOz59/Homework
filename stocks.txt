Sub stock_analysis():

Set results_sheet = Worksheets("Results")
Dim total As Double
Dim i As Long
Dim change As Single
Dim j As Integer
Dim starter As Long
Dim rowCounter As Long
Dim percentageChange As Single
Dim days As Integer
Dim dailyChange As Single
Dim avgChange As Single

j = 0
total = 0
change = 0
start = 2
dailyChange = 0


For Each ws In Worksheets



rowCounter = Cells(Rows.Count, "A").End(xlUp).Row

For i = 2 To rowCounter



    If Cells(i + 1, 1).Value <> Cells(i, 1).Value Then

        total = total + Cells(i, 7).Value
        change = (Cells(i, 6) - Cells(start, 3))
        percentageChange = Round((change / Cells(start, 3) * 100), 2)
        dailyChange = dailyChange + (Cells(i, 4) - Cells(i, 5))

       
        days = (i - start) + 1
        avgChange = dailyChange / days

   
        starter = i + 1

   
        Range("I" & 2 + j).Value = Cells(i, 1).Value
        Range("J" & 2 + j).Value = Round(change, 2)
        Range("K" & 2 + j).Value = "%" & percentageChange
        Range("D" & 2 + j).Value = avgChange
        Range("L" & 2 + j).Value = total

     
        Select Case change
            Case Is > 0
               Range("J" & 2 + j).Interior.ColorIndex = 4
            Case Is < 0
                Range("J" & 2 + j).Interior.ColorIndex = 3
            Case Else
                Range("J" & 2 + j).Interior.ColorIndex = 0
        End Select



        ' resetting var
        total = 0
        change = 0
        j = j + 1
        days = 0
        dailyChange = 0

   
    Else
        total = total + Cells(i, 7).Value
        change = change + (Cells(i, 6) - Cells(i, 3))

        
        dailyChange = dailyChange + (Cells(i, 4) - Cells(i, 5))


    End If




Next i

    lastRow = results_sheet.Cells(Rows.Count, "A").End(xlUp).Row + 1

       
    results_sheet.Range("A" & lastRow & ":G" & 30 + lastRow).Value = ws.Range("A2:G32").Value

    Next ws
    



End Sub
